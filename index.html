My favorite piece of Swift code: the first line of script’s .swift file should be a “shebang”, which points to the swift executable and we do not have to compile the code in case of small test program
    
#!/usr/bin/swift

var fibMemo: [UInt: UInt] = [0: 0, 1: 1]   // our old base cases
func fibonacci(n: UInt) -> UInt {
    if let result = fibMemo[n] {   // our new base case
        return result
    } else {
        fibMemo[n] = fib3(n: n - 1) + fib3(n: n - 2)   // memoization
    }
    return fibMemo[n]!
}

print(fibonacci(10))
